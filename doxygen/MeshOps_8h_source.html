145 // Get the number of processes that participate in each group
146 // induced by `meshAxes`.
147 template <typename MeshAxesRange, typename MeshShapeRange>
148 int64_t collectiveProcessGroupSize(MeshAxesRange &&meshAxes,
149  MeshShapeRange &&meshShape) {
150  int64_t res = 1;
151 
152  for (MeshAxis axis : meshAxes) {
153  auto axisSize = *(std::begin(meshShape) + axis);
154  if (ShapedType::isDynamic(axisSize)) {
155  return ShapedType::kDynamic;
156  }
157  res *= axisSize;
158  }
159 
160  return res;
161 }

163 template <typename MeshAxesRange>
164 int64_t collectiveProcessGroupSize(MeshAxesRange &&meshAxes, MeshOp mesh) {
165  return collectiveProcessGroupSize(std::forward<MeshAxesRange>(meshAxes),
166  mesh.getShape());
167 }

169 // Get the size of a sharded dimension.
170 inline int64_t shardDimension(int64_t dimSize, int64_t shardCount) {
171  if (ShapedType::isDynamic(dimSize) || ShapedType::isDynamic(shardCount))
172  return ShapedType::kDynamic;
173 
174  assert(dimSize % shardCount == 0);
175  return dimSize / shardCount;
176 }
177 
178 // Get the size of an unsharded dimension.
179 inline int64_t gatherDimension(int64_t dimSize, int64_t shardCount) {
180  if (ShapedType::isDynamic(dimSize) || ShapedType::isDynamic(shardCount))
181  return ShapedType::kDynamic;
182 
183  return dimSize * shardCount;
184 }
