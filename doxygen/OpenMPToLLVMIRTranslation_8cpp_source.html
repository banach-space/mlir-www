94 /// Find the insertion point for allocas given the current insertion point for
95 /// normal operations in the builder.
96 static llvm::OpenMPIRBuilder::InsertPointTy
97 findAllocaInsertPoint(llvm::IRBuilderBase &builder,
98  const LLVM::ModuleTranslation &moduleTranslation) {
99  // If there is an alloca insertion point on stack, i.e. we are in a nested
100  // operation and a specific point was provided by some surrounding operation,
101  // use it.
102  llvm::OpenMPIRBuilder::InsertPointTy allocaInsertPoint;
103  WalkResult walkResult = moduleTranslation.stackWalk<OpenMPAllocaStackFrame>(
104  [&](const OpenMPAllocaStackFrame &frame) {
105  allocaInsertPoint = frame.allocaInsertPoint;
106  return WalkResult::interrupt();
107  });
108  if (walkResult.wasInterrupted())
109  return allocaInsertPoint;
110 
111  // Otherwise, insert to the entry block of the surrounding function.
112  // If the current IRBuilder InsertPoint is the function's entry, it cannot
113  // also be used for alloca insertion which would result in insertion order
114  // confusion. Create a new BasicBlock for the Builder and use the entry block
115  // for the allocs.
116  // TODO: Create a dedicated alloca BasicBlock at function creation such that
117  // we do not need to move the current InertPoint here.
118  if (builder.GetInsertBlock() ==
119  &builder.GetInsertBlock()->getParent()->getEntryBlock()) {
120  assert(builder.GetInsertPoint() == builder.GetInsertBlock()->end() &&
121  "Assuming end of basic block");
122  llvm::BasicBlock *entryBB = llvm::BasicBlock::Create(
123  builder.getContext(), "entry", builder.GetInsertBlock()->getParent(),
124  builder.GetInsertBlock()->getNextNode());
125  builder.CreateBr(entryBB);
126  builder.SetInsertPoint(entryBB);
127  }
128 
129  llvm::BasicBlock &funcEntryBlock =
130  builder.GetInsertBlock()->getParent()->getEntryBlock();
131  return llvm::OpenMPIRBuilder::InsertPointTy(
132  &funcEntryBlock, funcEntryBlock.getFirstInsertionPt());
133 }

135 /// Converts the given region that appears within an OpenMP dialect operation to
136 /// LLVM IR, creating a branch from the `sourceBlock` to the entry block of the
137 /// region, and a branch from any block with an successor-less OpenMP terminator
138 /// to `continuationBlock`. Populates `continuationBlockPHIs` with the PHI nodes
139 /// of the continuation block if provided.
140 static llvm::BasicBlock *convertOmpOpRegions(
141  Region &region, StringRef blockName, llvm::IRBuilderBase &builder,
142  LLVM::ModuleTranslation &moduleTranslation, LogicalResult &bodyGenStatus,
143  SmallVectorImpl<llvm::PHINode *> *continuationBlockPHIs = nullptr) {
144  llvm::BasicBlock *continuationBlock =
145  splitBB(builder, true, "omp.region.cont");
146  llvm::BasicBlock *sourceBlock = builder.GetInsertBlock();
147 
148  llvm::LLVMContext &llvmContext = builder.getContext();
149  for (Block &bb : region) {
150  llvm::BasicBlock *llvmBB = llvm::BasicBlock::Create(
151  llvmContext, blockName, builder.GetInsertBlock()->getParent(),
152  builder.GetInsertBlock()->getNextNode());
153  moduleTranslation.mapBlock(&bb, llvmBB);
154  }
155 
156  llvm::Instruction *sourceTerminator = sourceBlock->getTerminator();
157 
158  // Terminators (namely YieldOp) may be forwarding values to the region that
159  // need to be available in the continuation block. Collect the types of these
160  // operands in preparation of creating PHI nodes.
161  SmallVector<llvm::Type *> continuationBlockPHITypes;
162  bool operandsProcessed = false;
163  unsigned numYields = 0;
164  for (Block &bb : region.getBlocks()) {
165  if (omp::YieldOp yield = dyn_cast<omp::YieldOp>(bb.getTerminator())) {
166  if (!operandsProcessed) {
167  for (unsigned i = 0, e = yield->getNumOperands(); i < e; ++i) {
168  continuationBlockPHITypes.push_back(
169  moduleTranslation.convertType(yield->getOperand(i).getType()));
170  }
171  operandsProcessed = true;
172  } else {
173  assert(continuationBlockPHITypes.size() == yield->getNumOperands() &&
174  "mismatching number of values yielded from the region");
175  for (unsigned i = 0, e = yield->getNumOperands(); i < e; ++i) {
176  llvm::Type *operandType =
177  moduleTranslation.convertType(yield->getOperand(i).getType());
178  (void)operandType;
179  assert(continuationBlockPHITypes[i] == operandType &&
180  "values of mismatching types yielded from the region");
181  }
182  }
183  numYields++;
184  }
185  }
186 
187  // Insert PHI nodes in the continuation block for any values forwarded by the
188  // terminators in this region.
189  if (!continuationBlockPHITypes.empty())
190  assert(
191  continuationBlockPHIs &&
192  "expected continuation block PHIs if converted regions yield values");
193  if (continuationBlockPHIs) {
194  llvm::IRBuilderBase::InsertPointGuard guard(builder);
195  continuationBlockPHIs->reserve(continuationBlockPHITypes.size());
196  builder.SetInsertPoint(continuationBlock, continuationBlock->begin());
197  for (llvm::Type *ty : continuationBlockPHITypes)
198  continuationBlockPHIs->push_back(builder.CreatePHI(ty, numYields));
199  }
200 
201  // Convert blocks one by one in topological order to ensure
202  // defs are converted before uses.
203  SetVector<Block *> blocks = getBlocksSortedByDominance(region);
204  for (Block *bb : blocks) {
205  llvm::BasicBlock *llvmBB = moduleTranslation.lookupBlock(bb);
206  // Retarget the branch of the entry block to the entry block of the
207  // converted region (regions are single-entry).
208  if (bb->isEntryBlock()) {
209  assert(sourceTerminator->getNumSuccessors() == 1 &&
210  "provided entry block has multiple successors");
211  assert(sourceTerminator->getSuccessor(0) == continuationBlock &&
212  "ContinuationBlock is not the successor of the entry block");
213  sourceTerminator->setSuccessor(0, llvmBB);
214  }
215 
216  llvm::IRBuilderBase::InsertPointGuard guard(builder);
217  if (failed(
218  moduleTranslation.convertBlock(*bb, bb->isEntryBlock(), builder))) {
219  bodyGenStatus = failure();
220  return continuationBlock;
221  }
222 
223  // Special handling for `omp.yield` and `omp.terminator` (we may have more
224  // than one): they return the control to the parent OpenMP dialect operation
225  // so replace them with the branch to the continuation block. We handle this
226  // here to avoid relying inter-function communication through the
227  // ModuleTranslation class to set up the correct insertion point. This is
228  // also consistent with MLIR's idiom of handling special region terminators
229  // in the same code that handles the region-owning operation.
230  Operation *terminator = bb->getTerminator();
231  if (isa<omp::TerminatorOp, omp::YieldOp>(terminator)) {
232  builder.CreateBr(continuationBlock);
233 
234  for (unsigned i = 0, e = terminator->getNumOperands(); i < e; ++i)
235  (*continuationBlockPHIs)[i]->addIncoming(
236  moduleTranslation.lookupValue(terminator->getOperand(i)), llvmBB);
237  }
238  }
239  // After all blocks have been traversed and values mapped, connect the PHI
240  // nodes to the results of preceding blocks.
241  LLVM::detail::connectPHINodes(region, moduleTranslation);
242 
243  // Remove the blocks and values defined in this region from the mapping since
244  // they are not visible outside of this region. This allows the same region to
245  // be converted several times, that is cloned, without clashes, and slightly
246  // speeds up the lookups.
247  moduleTranslation.forgetMapping(region);
248 
249  return continuationBlock;
250 }

267 /// Converts an OpenMP 'masked' operation into LLVM IR using OpenMPIRBuilder.
268 static LogicalResult
269 convertOmpMasked(Operation &opInst, llvm::IRBuilderBase &builder,
270  LLVM::ModuleTranslation &moduleTranslation) {
271  auto maskedOp = cast<omp::MaskedOp>(opInst);
272  using InsertPointTy = llvm::OpenMPIRBuilder::InsertPointTy;
273  // TODO: support error propagation in OpenMPIRBuilder and use it instead of
274  // relying on captured variables.
275  LogicalResult bodyGenStatus = success();
276 
277  auto bodyGenCB = [&](InsertPointTy allocaIP, InsertPointTy codeGenIP) {
278  // MaskedOp has only one region associated with it.
279  auto &region = maskedOp.getRegion();
280  builder.restoreIP(codeGenIP);
281  convertOmpOpRegions(region, "omp.masked.region", builder, moduleTranslation,
282  bodyGenStatus);
283  };
284 
285  // TODO: Perform finalization actions for variables. This has to be
286  // called for variables which have destructors/finalizers.
287  auto finiCB = [&](InsertPointTy codeGenIP) {};
288 
289  llvm::Value *filterVal = nullptr;
290  if (auto filterVar = maskedOp.getFilteredThreadId()) {
291  filterVal = moduleTranslation.lookupValue(filterVar);
292  } else {
293  llvm::LLVMContext &llvmContext = builder.getContext();
294  filterVal =
295  llvm::ConstantInt::get(llvm::Type::getInt32Ty(llvmContext), /*V=*/0);
296  }
297  assert(filterVal != nullptr);
298  llvm::OpenMPIRBuilder::LocationDescription ompLoc(builder);
299  builder.restoreIP(moduleTranslation.getOpenMPBuilder()->createMasked(
300  ompLoc, bodyGenCB, finiCB, filterVal));
301  return success();
302 }

391 /// Translates the blocks contained in the given region and appends them to at
392 /// the current insertion point of `builder`. The operations of the entry block
393 /// are appended to the current insertion block. If set, `continuationBlockArgs`
394 /// is populated with translated values that correspond to the values
395 /// omp.yield'ed from the region.
396 static LogicalResult inlineConvertOmpRegions(
397  Region &region, StringRef blockName, llvm::IRBuilderBase &builder,
398  LLVM::ModuleTranslation &moduleTranslation,
399  SmallVectorImpl<llvm::Value *> *continuationBlockArgs = nullptr) {
400  if (region.empty())
401  return success();
402 
403  // Special case for single-block regions that don't create additional blocks:
404  // insert operations without creating additional blocks.
405  if (llvm::hasSingleElement(region)) {
406  llvm::Instruction *potentialTerminator =
407  builder.GetInsertBlock()->empty() ? nullptr
408  : &builder.GetInsertBlock()->back();
409 
410  if (potentialTerminator && potentialTerminator->isTerminator())
411  potentialTerminator->removeFromParent();
412  moduleTranslation.mapBlock(&region.front(), builder.GetInsertBlock());
413 
414  if (failed(moduleTranslation.convertBlock(
415  region.front(), /*ignoreArguments=*/true, builder)))
416  return failure();
417 
418  // The continuation arguments are simply the translated terminator operands.
419  if (continuationBlockArgs)
420  llvm::append_range(
421  *continuationBlockArgs,
422  moduleTranslation.lookupValues(region.front().back().getOperands()));
423 
424  // Drop the mapping that is no longer necessary so that the same region can
425  // be processed multiple times.
426  moduleTranslation.forgetMapping(region);
427 
428  if (potentialTerminator && potentialTerminator->isTerminator()) {
429  llvm::BasicBlock *block = builder.GetInsertBlock();
430  if (block->empty()) {
431  // this can happen for really simple reduction init regions e.g.
432  // %0 = llvm.mlir.constant(0 : i32) : i32
433  // omp.yield(%0 : i32)
434  // because the llvm.mlir.constant (MLIR op) isn't converted into any
435  // llvm op
436  potentialTerminator->insertInto(block, block->begin());
437  } else {
438  potentialTerminator->insertAfter(&block->back());
439  }
440  }
441 
442  return success();
443  }
444 
445  LogicalResult bodyGenStatus = success();
446  SmallVector<llvm::PHINode *> phis;
447  llvm::BasicBlock *continuationBlock = convertOmpOpRegions(
448  region, blockName, builder, moduleTranslation, bodyGenStatus, &phis);
449  if (failed(bodyGenStatus))
450  return failure();
451  if (continuationBlockArgs)
452  llvm::append_range(*continuationBlockArgs, phis);
453  builder.SetInsertPoint(continuationBlock,
454  continuationBlock->getFirstInsertionPt());
455  return success();
456 }

470 /// Create an OpenMPIRBuilder-compatible reduction generator for the given
471 /// reduction declaration. The generator uses `builder` but ignores its
472 /// insertion point.
473 static OwningReductionGen
474 makeReductionGen(omp::DeclareReductionOp decl, llvm::IRBuilderBase &builder,
475  LLVM::ModuleTranslation &moduleTranslation) {
476  // The lambda is mutable because we need access to non-const methods of decl
477  // (which aren't actually mutating it), and we must capture decl by-value to
478  // avoid the dangling reference after the parent function returns.
479  OwningReductionGen gen =
480  [&, decl](llvm::OpenMPIRBuilder::InsertPointTy insertPoint,
481  llvm::Value *lhs, llvm::Value *rhs,
482  llvm::Value *&result) mutable {
483  Region &reductionRegion = decl.getReductionRegion();
484  moduleTranslation.mapValue(reductionRegion.front().getArgument(0), lhs);
485  moduleTranslation.mapValue(reductionRegion.front().getArgument(1), rhs);
486  builder.restoreIP(insertPoint);
487  SmallVector<llvm::Value *> phis;
488  if (failed(inlineConvertOmpRegions(reductionRegion,
489  "omp.reduction.nonatomic.body",
490  builder, moduleTranslation, &phis)))
491  return llvm::OpenMPIRBuilder::InsertPointTy();
492  assert(phis.size() == 1);
493  result = phis[0];
494  return builder.saveIP();
495  };
496  return gen;
497 }

531 /// Converts an OpenMP 'ordered' operation into LLVM IR using OpenMPIRBuilder.
532 static LogicalResult
533 convertOmpOrdered(Operation &opInst, llvm::IRBuilderBase &builder,
534  LLVM::ModuleTranslation &moduleTranslation) {
535  auto orderedOp = cast<omp::OrderedOp>(opInst);
536 
537  omp::ClauseDepend dependType = *orderedOp.getDoacrossDependType();
538  bool isDependSource = dependType == omp::ClauseDepend::dependsource;
539  unsigned numLoops = *orderedOp.getDoacrossNumLoops();
540  SmallVector<llvm::Value *> vecValues =
541  moduleTranslation.lookupValues(orderedOp.getDoacrossDependVars());
542 
543  size_t indexVecValues = 0;
544  while (indexVecValues < vecValues.size()) {
545  SmallVector<llvm::Value *> storeValues;
546  storeValues.reserve(numLoops);
547  for (unsigned i = 0; i < numLoops; i++) {
548  storeValues.push_back(vecValues[indexVecValues]);
549  indexVecValues++;
550  }
551  llvm::OpenMPIRBuilder::InsertPointTy allocaIP =
552  findAllocaInsertPoint(builder, moduleTranslation);
553  llvm::OpenMPIRBuilder::LocationDescription ompLoc(builder);
554  builder.restoreIP(moduleTranslation.getOpenMPBuilder()->createOrderedDepend(
555  ompLoc, allocaIP, numLoops, storeValues, ".cnt.addr", isDependSource));
556  }
557  return success();
558 }

606 /// Allocate space for privatized reduction variables.
607 /// `deferredStores` contains information to create store operations which needs
608 /// to be inserted after all allocas
609 template <typename T>
610 static LogicalResult
611 allocReductionVars(T loop, ArrayRef<BlockArgument> reductionArgs,
612  llvm::IRBuilderBase &builder,
613  LLVM::ModuleTranslation &moduleTranslation,
614  llvm::OpenMPIRBuilder::InsertPointTy &allocaIP,
615  SmallVectorImpl<omp::DeclareReductionOp> &reductionDecls,
616  SmallVectorImpl<llvm::Value *> &privateReductionVariables,
617  DenseMap<Value, llvm::Value *> &reductionVariableMap,
618  SmallVectorImpl<DeferredStore> &deferredStores,
619  llvm::ArrayRef<bool> isByRefs) {
620  llvm::IRBuilderBase::InsertPointGuard guard(builder);
621  builder.SetInsertPoint(allocaIP.getBlock()->getTerminator());
622 
623  // delay creating stores until after all allocas
624  deferredStores.reserve(loop.getNumReductionVars());
625 
626  for (std::size_t i = 0; i < loop.getNumReductionVars(); ++i) {
627  Region &allocRegion = reductionDecls[i].getAllocRegion();
628  if (isByRefs[i]) {
629  if (allocRegion.empty())
630  continue;
631 
632  SmallVector<llvm::Value *, 1> phis;
633  if (failed(inlineConvertOmpRegions(allocRegion, "omp.reduction.alloc",
634  builder, moduleTranslation, &phis)))
635  return failure();
636  assert(phis.size() == 1 && "expected one allocation to be yielded");
637 
638  builder.SetInsertPoint(allocaIP.getBlock()->getTerminator());
639 
640  // Allocate reduction variable (which is a pointer to the real reduction
641  // variable allocated in the inlined region)
642  llvm::Value *var = builder.CreateAlloca(
643  moduleTranslation.convertType(reductionDecls[i].getType()));
644  deferredStores.emplace_back(phis[0], var);
645 
646  privateReductionVariables[i] = var;
647  moduleTranslation.mapValue(reductionArgs[i], phis[0]);
648  reductionVariableMap.try_emplace(loop.getReductionVars()[i], phis[0]);
649  } else {
650  assert(allocRegion.empty() &&
651  "allocaction is implicit for by-val reduction");
652  llvm::Value *var = builder.CreateAlloca(
653  moduleTranslation.convertType(reductionDecls[i].getType()));
654  moduleTranslation.mapValue(reductionArgs[i], var);
655  privateReductionVariables[i] = var;
656  reductionVariableMap.try_emplace(loop.getReductionVars()[i], var);
657  }
658  }
659 
660  return success();
661 }

689 static void collectReductionInfo(

691  LLVM::ModuleTranslation &moduleTranslation,

723 /// handling of DeclareReductionOp's cleanup region
724 static LogicalResult
725 inlineOmpRegionCleanup(llvm::SmallVectorImpl<Region *> &cleanupRegions,
726  llvm::ArrayRef<llvm::Value *> privateVariables,
727  LLVM::ModuleTranslation &moduleTranslation,
728  llvm::IRBuilderBase &builder, StringRef regionName,
729  bool shouldLoadCleanupRegionArg = true) {
730  for (auto [i, cleanupRegion] : llvm::enumerate(cleanupRegions)) {
731  if (cleanupRegion->empty())
732  continue;
733 
734  // map the argument to the cleanup region
735  Block &entry = cleanupRegion->front();
736 
737  llvm::Instruction *potentialTerminator =
738  builder.GetInsertBlock()->empty() ? nullptr
739  : &builder.GetInsertBlock()->back();
740  if (potentialTerminator && potentialTerminator->isTerminator())
741  builder.SetInsertPoint(potentialTerminator);
742  llvm::Value *privateVarValue =
743  shouldLoadCleanupRegionArg
744  ? builder.CreateLoad(
745  moduleTranslation.convertType(entry.getArgument(0).getType()),
746  privateVariables[i])
747  : privateVariables[i];
748 
749  moduleTranslation.mapValue(entry.getArgument(0), privateVarValue);
750 
751  if (failed(inlineConvertOmpRegions(*cleanupRegion, regionName, builder,
752  moduleTranslation)))
753  return failure();
754 
755  // clear block argument mapping in case it needs to be re-created with a
756  // different source for another use of the same reduction decl
757  moduleTranslation.forgetMapping(*cleanupRegion);
758  }
759  return success();
760 }

762 // TODO: not used by ParallelOp
763 template <class OP>
764 static LogicalResult createReductionsAndCleanup(
765  OP op, llvm::IRBuilderBase &builder,
766  LLVM::ModuleTranslation &moduleTranslation,
767  llvm::OpenMPIRBuilder::InsertPointTy &allocaIP,
768  SmallVectorImpl<omp::DeclareReductionOp> &reductionDecls,
769  ArrayRef<llvm::Value *> privateReductionVariables, ArrayRef<bool> isByRef) {
770  // Process the reductions if required.
771  if (op.getNumReductionVars() == 0)
772  return success();
773 
774  llvm::OpenMPIRBuilder *ompBuilder = moduleTranslation.getOpenMPBuilder();
775 
776  // Create the reduction generators. We need to own them here because
777  // ReductionInfo only accepts references to the generators.
778  SmallVector<OwningReductionGen> owningReductionGens;
779  SmallVector<OwningAtomicReductionGen> owningAtomicReductionGens;
780  SmallVector<llvm::OpenMPIRBuilder::ReductionInfo> reductionInfos;
781  collectReductionInfo(op, builder, moduleTranslation, reductionDecls,
782  owningReductionGens, owningAtomicReductionGens,
783  privateReductionVariables, reductionInfos);
784 
785  // The call to createReductions below expects the block to have a
786  // terminator. Create an unreachable instruction to serve as terminator
787  // and remove it later.
788  llvm::UnreachableInst *tempTerminator = builder.CreateUnreachable();
789  builder.SetInsertPoint(tempTerminator);
790  llvm::OpenMPIRBuilder::InsertPointTy contInsertPoint =
791  ompBuilder->createReductions(builder.saveIP(), allocaIP, reductionInfos,
792  isByRef, op.getNowait());
793  if (!contInsertPoint.getBlock())
794  return op->emitOpError() << "failed to convert reductions";
795  auto nextInsertionPoint =
796  ompBuilder->createBarrier(contInsertPoint, llvm::omp::OMPD_for);
797  tempTerminator->eraseFromParent();
798  builder.restoreIP(nextInsertionPoint);
799 
800  // after the construct, deallocate private reduction variables
801  SmallVector<Region *> reductionRegions;
802  llvm::transform(reductionDecls, std::back_inserter(reductionRegions),
803  [](omp::DeclareReductionOp reductionDecl) {
804  return &reductionDecl.getCleanupRegion();
805  });
806  return inlineOmpRegionCleanup(reductionRegions, privateReductionVariables,
807  moduleTranslation, builder,
808  "omp.reduction.cleanup");
809  return success();
810 }

818 // TODO: not used by omp.parallel
819 template <typename OP>
820 static LogicalResult allocAndInitializeReductionVars(
821  OP op, ArrayRef<BlockArgument> reductionArgs, llvm::IRBuilderBase &builder,
822  LLVM::ModuleTranslation &moduleTranslation,
823  llvm::OpenMPIRBuilder::InsertPointTy &allocaIP,
824  SmallVectorImpl<omp::DeclareReductionOp> &reductionDecls,
825  SmallVectorImpl<llvm::Value *> &privateReductionVariables,
826  DenseMap<Value, llvm::Value *> &reductionVariableMap,
827  llvm::ArrayRef<bool> isByRef) {
828  if (op.getNumReductionVars() == 0)
829  return success();
830 
831  SmallVector<DeferredStore> deferredStores;
832 
833  if (failed(allocReductionVars(op, reductionArgs, builder, moduleTranslation,
834  allocaIP, reductionDecls,
835  privateReductionVariables, reductionVariableMap,
836  deferredStores, isByRef)))
837  return failure();
838 
839  // store result of the alloc region to the allocated pointer to the real
840  // reduction variable
841  for (auto [data, addr] : deferredStores)
842  builder.CreateStore(data, addr);
843 
844  // Before the loop, store the initial values of reductions into reduction
845  // variables. Although this could be done after allocas, we don't want to mess
846  // up with the alloca insertion point.
847  for (unsigned i = 0; i < op.getNumReductionVars(); ++i) {
848  SmallVector<llvm::Value *, 1> phis;
849 
850  // map block argument to initializer region
851  mapInitializationArgs(op, moduleTranslation, reductionDecls,
852  reductionVariableMap, i);
853 
854  if (failed(inlineConvertOmpRegions(reductionDecls[i].getInitializerRegion(),
855  "omp.reduction.neutral", builder,
856  moduleTranslation, &phis)))
857  return failure();
858  assert(phis.size() == 1 && "expected one value to be yielded from the "
859  "reduction neutral element declaration region");
860  if (isByRef[i]) {
861  if (!reductionDecls[i].getAllocRegion().empty())
862  // done in allocReductionVars
863  continue;
864 
865  // TODO: this path can be removed once all users of by-ref are updated to
866  // use an alloc region
867 
868  // Allocate reduction variable (which is a pointer to the real reduction
869  // variable allocated in the inlined region)
870  llvm::Value *var = builder.CreateAlloca(
871  moduleTranslation.convertType(reductionDecls[i].getType()));
872  // Store the result of the inlined region to the allocated reduction var
873  // ptr
874  builder.CreateStore(phis[0], var);
875 
876  privateReductionVariables[i] = var;
877  moduleTranslation.mapValue(reductionArgs[i], phis[0]);
878  reductionVariableMap.try_emplace(op.getReductionVars()[i], phis[0]);
879  } else {
880  // for by-ref case the store is inside of the reduction region
881  builder.CreateStore(phis[0], privateReductionVariables[i]);
882  // the rest was handled in allocByValReductionVars
883  }
884 
885  // forget the mapping for the initializer region because we might need a
886  // different mapping if this reduction declaration is re-used for a
887  // different variable
888  moduleTranslation.forgetMapping(reductionDecls[i].getInitializerRegion());
889  }
890 
891  return success();
892 }

894 static LogicalResult
895 convertOmpSections(Operation &opInst, llvm::IRBuilderBase &builder,
896  LLVM::ModuleTranslation &moduleTranslation) {
897  using InsertPointTy = llvm::OpenMPIRBuilder::InsertPointTy;
898  using StorableBodyGenCallbackTy =
899  llvm::OpenMPIRBuilder::StorableBodyGenCallbackTy;
900 
901  auto sectionsOp = cast<omp::SectionsOp>(opInst);
902 
903  // TODO: Support the following clauses: private, firstprivate, lastprivate,
904  // allocate
905  if (!sectionsOp.getAllocateVars().empty() ||
906  !sectionsOp.getAllocatorVars().empty() ||
907  !sectionsOp.getPrivateVars().empty() || sectionsOp.getPrivateSyms())
908  return opInst.emitError("unhandled clauses for translation to LLVM IR");
909 
910  llvm::ArrayRef<bool> isByRef = getIsByRef(sectionsOp.getReductionByref());
911  assert(isByRef.size() == sectionsOp.getNumReductionVars());
912 
913  SmallVector<omp::DeclareReductionOp> reductionDecls;
914  collectReductionDecls(sectionsOp, reductionDecls);
915  llvm::OpenMPIRBuilder::InsertPointTy allocaIP =
916  findAllocaInsertPoint(builder, moduleTranslation);
917 
918  SmallVector<llvm::Value *> privateReductionVariables(
919  sectionsOp.getNumReductionVars());
920  DenseMap<Value, llvm::Value *> reductionVariableMap;
921 
922  MutableArrayRef<BlockArgument> reductionArgs =
923  cast<omp::BlockArgOpenMPOpInterface>(opInst).getReductionBlockArgs();
924 
925  if (failed(allocAndInitializeReductionVars(
926  sectionsOp, reductionArgs, builder, moduleTranslation, allocaIP,
927  reductionDecls, privateReductionVariables, reductionVariableMap,
928  isByRef)))
929  return failure();
930 
931  // Store the mapping between reduction variables and their private copies on
932  // ModuleTranslation stack. It can be then recovered when translating
933  // omp.reduce operations in a separate call.
934  LLVM::ModuleTranslation::SaveStack<OpenMPVarMappingStackFrame> mappingGuard(
935  moduleTranslation, reductionVariableMap);
936 
937  LogicalResult bodyGenStatus = success();
938  SmallVector<StorableBodyGenCallbackTy> sectionCBs;
939 
940  for (Operation &op : *sectionsOp.getRegion().begin()) {
941  auto sectionOp = dyn_cast<omp::SectionOp>(op);
942  if (!sectionOp) // omp.terminator
943  continue;
944 
945  Region &region = sectionOp.getRegion();
946  auto sectionCB = [&sectionsOp, &region, &builder, &moduleTranslation,
947  &bodyGenStatus](InsertPointTy allocaIP,
948  InsertPointTy codeGenIP) {
949  builder.restoreIP(codeGenIP);
950 
951  // map the omp.section reduction block argument to the omp.sections block
952  // arguments
953  // TODO: this assumes that the only block arguments are reduction
954  // variables
955  assert(region.getNumArguments() ==
956  sectionsOp.getRegion().getNumArguments());
957  for (auto [sectionsArg, sectionArg] : llvm::zip_equal(
958  sectionsOp.getRegion().getArguments(), region.getArguments())) {
959  llvm::Value *llvmVal = moduleTranslation.lookupValue(sectionsArg);
960  assert(llvmVal);
961  moduleTranslation.mapValue(sectionArg, llvmVal);
962  }
963 
964  convertOmpOpRegions(region, "omp.section.region", builder,
965  moduleTranslation, bodyGenStatus);
966  };
967  sectionCBs.push_back(sectionCB);
968  }
969 
970  // No sections within omp.sections operation - skip generation. This situation
971  // is only possible if there is only a terminator operation inside the
972  // sections operation
973  if (sectionCBs.empty())
974  return success();
975 
976  assert(isa<omp::SectionOp>(*sectionsOp.getRegion().op_begin()));
977 
978  // TODO: Perform appropriate actions according to the data-sharing
979  // attribute (shared, private, firstprivate, ...) of variables.
980  // Currently defaults to shared.
981  auto privCB = [&](InsertPointTy, InsertPointTy codeGenIP, llvm::Value &,
982  llvm::Value &vPtr,
983  llvm::Value *&replacementValue) -> InsertPointTy {
984  replacementValue = &vPtr;
985  return codeGenIP;
986  };
987 
988  // TODO: Perform finalization actions for variables. This has to be
989  // called for variables which have destructors/finalizers.
990  auto finiCB = [&](InsertPointTy codeGenIP) {};
991 
992  allocaIP = findAllocaInsertPoint(builder, moduleTranslation);
993  llvm::OpenMPIRBuilder::LocationDescription ompLoc(builder);
994  builder.restoreIP(moduleTranslation.getOpenMPBuilder()->createSections(
995  ompLoc, allocaIP, sectionCBs, privCB, finiCB, false,
996  sectionsOp.getNowait()));
997 
998  if (failed(bodyGenStatus))
999  return bodyGenStatus;
1000 
1001  // Process the reductions if required.
1002  return createReductionsAndCleanup(sectionsOp, builder, moduleTranslation,
1003  allocaIP, reductionDecls,
1004  privateReductionVariables, isByRef);
1005 }

1098  case mlir::omp::ClauseTaskDepend::taskdependout:
1099  case mlir::omp::ClauseTaskDepend::taskdependinout:

1101  break;

1104  llvm::OpenMPIRBuilder::DependData dd(type, depVal->getType(), depVal);

1178 static LogicalResult
1179 convertOmpWsloop(Operation &opInst, llvm::IRBuilderBase &builder,
1180  LLVM::ModuleTranslation &moduleTranslation) {

1182  if (!wsloopOp.getAllocateVars().empty() ||
1183  !wsloopOp.getAllocatorVars().empty() ||
1184  !wsloopOp.getPrivateVars().empty() || wsloopOp.getPrivateSyms())
1185  return opInst.emitError("unhandled clauses for translation to LLVM IR");
1186 
1187  // FIXME: Here any other nested wrappers (e.g. omp.simd) are skipped, so
1188  // codegen for composite constructs like 'DO/FOR SIMD' will be the same as for
1189  // 'DO/FOR'.

1191 
1192  llvm::ArrayRef<bool> isByRef = getIsByRef(wsloopOp.getReductionByref());
1193  assert(isByRef.size() == wsloopOp.getNumReductionVars());
1194 
1195  // Static is the default.
1196  auto schedule =
1197  wsloopOp.getScheduleKind().value_or(omp::ClauseScheduleKind::Static);
1198 
1199  // Find the loop configuration.
1200  llvm::Value *step = moduleTranslation.lookupValue(loopOp.getLoopSteps()[0]);
1201  llvm::Type *ivType = step->getType();
1202  llvm::Value *chunk = nullptr;
1203  if (wsloopOp.getScheduleChunk()) {
1204  llvm::Value *chunkVar =
1205  moduleTranslation.lookupValue(wsloopOp.getScheduleChunk());
1206  chunk = builder.CreateSExtOrTrunc(chunkVar, ivType);
1207  }
1208

1210  collectReductionDecls(wsloopOp, reductionDecls);
1211  llvm::OpenMPIRBuilder::InsertPointTy allocaIP =
1212  findAllocaInsertPoint(builder, moduleTranslation);

1215  wsloopOp.getNumReductionVars());

1217 
1218  MutableArrayRef<BlockArgument> reductionArgs =
1219  cast<omp::BlockArgOpenMPOpInterface>(opInst).getReductionBlockArgs();
1220

1222  wsloopOp, reductionArgs, builder, moduleTranslation, allocaIP,
1223  reductionDecls, privateReductionVariables, reductionVariableMap,
1224  isByRef)))
1225  return failure();

1227  // Store the mapping between reduction variables and their private copies on
1228  // ModuleTranslation stack. It can be then recovered when translating
1229  // omp.reduce operations in a separate call.
1230  LLVM::ModuleTranslation::SaveStack<OpenMPVarMappingStackFrame> mappingGuard(
1231  moduleTranslation, reductionVariableMap);
1232 
1233  // Set up the source location value for OpenMP runtime.
1234  llvm::OpenMPIRBuilder::LocationDescription ompLoc(builder);
1235 
1236  // Generator of the canonical loop body.
1237  // TODO: support error propagation in OpenMPIRBuilder and use it instead of
1238  // relying on captured variables.
1239  SmallVector<llvm::CanonicalLoopInfo *> loopInfos;
1240  SmallVector<llvm::OpenMPIRBuilder::InsertPointTy> bodyInsertPoints;
1241  LogicalResult bodyGenStatus = success();

1243  // Make sure further conversions know about the induction variable.
1244  moduleTranslation.mapValue(
1245  loopOp.getRegion().front().getArgument(loopInfos.size()), iv);
1246 
1247  // Capture the body insertion point for use in nested loops. BodyIP of the
1248  // CanonicalLoopInfo always points to the beginning of the entry block of
1249  // the body.
1250  bodyInsertPoints.push_back(ip);
1251 
1252  if (loopInfos.size() != loopOp.getNumLoops() - 1)
1253  return;
1254 
1255  // Convert the body of the loop.
1256  builder.restoreIP(ip);
1257  convertOmpOpRegions(loopOp.getRegion(), "omp.wsloop.region", builder,
1258  moduleTranslation, bodyGenStatus);
1259  };

1261  // Delegate actual loop construction to the OpenMP IRBuilder.
1262  // TODO: this currently assumes omp.loop_nest is semantically similar to SCF
1263  // loop, i.e. it has a positive step, uses signed integer semantics.
1264  // Reconsider this code when the nested loop operation clearly supports more
1265  // cases.
1266  llvm::OpenMPIRBuilder *ompBuilder = moduleTranslation.getOpenMPBuilder();
1267  for (unsigned i = 0, e = loopOp.getNumLoops(); i < e; ++i) {
1268  llvm::Value *lowerBound =
1269  moduleTranslation.lookupValue(loopOp.getLoopLowerBounds()[i]);
1270  llvm::Value *upperBound =
1271  moduleTranslation.lookupValue(loopOp.getLoopUpperBounds()[i]);
1272  llvm::Value *step = moduleTranslation.lookupValue(loopOp.getLoopSteps()[i]);
1273 
1274  // Make sure loop trip count are emitted in the preheader of the outermost
1275  // loop at the latest so that they are all available for the new collapsed
1276  // loop will be created below.
1277  llvm::OpenMPIRBuilder::LocationDescription loc = ompLoc;
1278  llvm::OpenMPIRBuilder::InsertPointTy computeIP = ompLoc.IP;
1279  if (i != 0) {
1280  loc = llvm::OpenMPIRBuilder::LocationDescription(bodyInsertPoints.back());
1281  computeIP = loopInfos.front()->getPreheaderIP();
1282  }
1283  loopInfos.push_back(ompBuilder->createCanonicalLoop(
1284  loc, bodyGen, lowerBound, upperBound, step,
1285  /*IsSigned=*/true, loopOp.getLoopInclusive(), computeIP));
1286 
1287  if (failed(bodyGenStatus))
1288  return failure();
1289  }
1290 
1291  // Collapse loops. Store the insertion point because LoopInfos may get
1292  // invalidated.
1293  llvm::IRBuilderBase::InsertPoint afterIP = loopInfos.front()->getAfterIP();
1294  llvm::CanonicalLoopInfo *loopInfo =
1295  ompBuilder->collapseLoops(ompLoc.DL, loopInfos, {});
1296 
1297  allocaIP = findAllocaInsertPoint(builder, moduleTranslation);
1298 
1299  // TODO: Handle doacross loops when the ordered clause has a parameter.
1300  bool isOrdered = wsloopOp.getOrdered().has_value();
1301  std::optional<omp::ScheduleModifier> scheduleMod = wsloopOp.getScheduleMod();
1302  bool isSimd = wsloopOp.getScheduleSimd();
1303 
1304  ompBuilder->applyWorkshareLoop(
1305  ompLoc.DL, loopInfo, allocaIP, !wsloopOp.getNowait(),
1306  convertToScheduleKind(schedule), chunk, isSimd,
1307  scheduleMod == omp::ScheduleModifier::monotonic,
1308  scheduleMod == omp::ScheduleModifier::nonmonotonic, isOrdered);
1309 
1310  // Continue building IR after the loop. Note that the LoopInfo returned by
1311  // `collapseLoops` points inside the outermost loop and is intended for
1312  // potential further loop transformations. Use the insertion point stored
1313  // before collapsing loops instead.
1314  builder.restoreIP(afterIP);
1315 
1316  // Process the reductions if required.
1317  return createReductionsAndCleanup(wsloopOp, builder, moduleTranslation,
1318  allocaIP, reductionDecls,
1319  privateReductionVariables, isByRef);
1320 }

1387 static LogicalResult

1392  ArrayRef<bool> isByRef = getIsByRef(opInst.getReductionByref());
1393  assert(isByRef.size() == opInst.getNumReductionVars());
1394 
1395  // TODO: support error propagation in OpenMPIRBuilder and use it instead of
1396  // relying on captured variables.
1397  LogicalResult bodyGenStatus = success();
1398  llvm::OpenMPIRBuilder *ompBuilder = moduleTranslation.getOpenMPBuilder();
1399 
1400  // Collect reduction declarations
1401  SmallVector<omp::DeclareReductionOp> reductionDecls;
1402  collectReductionDecls(opInst, reductionDecls);
1403  SmallVector<llvm::Value *> privateReductionVariables(
1404  opInst.getNumReductionVars());

1406 
1407  auto bodyGenCB = [&](InsertPointTy allocaIP, InsertPointTy codeGenIP) {
1408  // Allocate reduction vars
1409  DenseMap<Value, llvm::Value *> reductionVariableMap;
1410 
1411  MutableArrayRef<BlockArgument> reductionArgs =
1412  cast<omp::BlockArgOpenMPOpInterface>(*opInst).getReductionBlockArgs();
1413 
1414  allocaIP =
1415  InsertPointTy(allocaIP.getBlock(),
1416  allocaIP.getBlock()->getTerminator()->getIterator());
1417 
1418  if (failed(allocReductionVars(
1419  opInst, reductionArgs, builder, moduleTranslation, allocaIP,
1420  reductionDecls, privateReductionVariables, reductionVariableMap,
1421  deferredStores, isByRef)))
1422  bodyGenStatus = failure();

1424  // Initialize reduction vars
1425  builder.restoreIP(allocaIP);
1426  llvm::BasicBlock *initBlock = splitBB(builder, true, "omp.reduction.init");
1427  allocaIP =
1428  InsertPointTy(allocaIP.getBlock(),
1429  allocaIP.getBlock()->getTerminator()->getIterator());

1431  for (unsigned i = 0; i < opInst.getNumReductionVars(); ++i) {
1432  if (isByRef[i]) {
1433  if (!reductionDecls[i].getAllocRegion().empty())
1434  continue;
1435 
1436  // TODO: remove after all users of by-ref are updated to use the alloc
1437  // region: Allocate reduction variable (which is a pointer to the real
1438  // reduciton variable allocated in the inlined region)
1439  byRefVars[i] = builder.CreateAlloca(
1440  moduleTranslation.convertType(reductionDecls[i].getType()));
1441  }
1442  }
1443 
1444  builder.SetInsertPoint(initBlock->getFirstNonPHIOrDbgOrAlloca());
1445 
1446  // insert stores deferred until after all allocas
1447  // these store the results of the alloc region into the allocation for the
1448  // pointer to the reduction variable
1449  for (auto [data, addr] : deferredStores)
1450  builder.CreateStore(data, addr);
1451 
1452  for (unsigned i = 0; i < opInst.getNumReductionVars(); ++i) {
1453  SmallVector<llvm::Value *> phis;
1454 
1455  // map the block argument
1456  mapInitializationArgs(opInst, moduleTranslation, reductionDecls,
1457  reductionVariableMap, i);
1458  if (failed(inlineConvertOmpRegions(
1459  reductionDecls[i].getInitializerRegion(), "omp.reduction.neutral",
1460  builder, moduleTranslation, &phis)))
1461  bodyGenStatus = failure();
1462  assert(phis.size() == 1 &&
1463  "expected one value to be yielded from the "
1464  "reduction neutral element declaration region");
1465

1467 
1468  if (isByRef[i]) {
1469  if (!reductionDecls[i].getAllocRegion().empty())
1470  continue;
1471 
1472  // TODO: remove after all users of by-ref are updated to use the alloc
1473 
1474  // Store the result of the inlined region to the allocated reduction var
1475  // ptr
1476  builder.CreateStore(phis[0], byRefVars[i]);
1477 
1478  privateReductionVariables[i] = byRefVars[i];
1479  moduleTranslation.mapValue(reductionArgs[i], phis[0]);
1480  reductionVariableMap.try_emplace(opInst.getReductionVars()[i], phis[0]);
1481  } else {
1482  // for by-ref case the store is inside of the reduction init region
1483  builder.CreateStore(phis[0], privateReductionVariables[i]);
1484  // the rest is done in allocByValReductionVars
1485  }
1486 
1487  // clear block argument mapping in case it needs to be re-created with a
1488  // different source for another use of the same reduction decl
1489  moduleTranslation.forgetMapping(reductionDecls[i].getInitializerRegion());
1490  }

1493  // ModuleTranslation stack. It can be then recovered when translating
1494  // omp.reduce operations in a separate call.
1495  LLVM::ModuleTranslation::SaveStack<OpenMPVarMappingStackFrame> mappingGuard(
1496  moduleTranslation, reductionVariableMap);
1497

1499  // nested regions.
1500  LLVM::ModuleTranslation::SaveStack<OpenMPAllocaStackFrame> frame(
1501  moduleTranslation, allocaIP);
1502

1504  builder.restoreIP(codeGenIP);
1505  auto regionBlock =
1506  convertOmpOpRegions(opInst.getRegion(), "omp.par.region", builder,
1507  moduleTranslation, bodyGenStatus);
1508 
1509  // Process the reductions if required.
1510  if (opInst.getNumReductionVars() > 0) {
1511  // Collect reduction info
1512  SmallVector<OwningReductionGen> owningReductionGens;
1513  SmallVector<OwningAtomicReductionGen> owningAtomicReductionGens;

1515  collectReductionInfo(opInst, builder, moduleTranslation, reductionDecls,

1517  privateReductionVariables, reductionInfos);
1518

1527  ompBuilder->createReductions(builder.saveIP(), allocaIP,

1540  SmallVector<llvm::Value *> llvmPrivateVars;
1541 
1542  // TODO: Perform appropriate actions according to the data-sharing
1543  // attribute (shared, private, firstprivate, ...) of variables.
1544  // Currently shared and private are supported.
1545  auto privCB = [&](InsertPointTy allocaIP, InsertPointTy codeGenIP,
1546  llvm::Value &, llvm::Value &llvmOmpRegionInput,
1547  llvm::Value *&llvmReplacementValue) -> InsertPointTy {
1548  llvmReplacementValue = &llvmOmpRegionInput;
1549 
1550  // If this is a private value, this lambda will return the corresponding
1551  // mlir value and its `PrivateClauseOp`. Otherwise, empty values are
1552  // returned.
1553  auto [mlirPrivVar, mlirPrivatizerClone] =
1554  [&]() -> std::pair<mlir::Value, omp::PrivateClauseOp> {
1555  if (!opInst.getPrivateVars().empty()) {
1556  auto mlirPrivVars = opInst.getPrivateVars();
1557  auto mlirPrivSyms = opInst.getPrivateSyms();
1558 
1559  // Try to find a privatizer that corresponds to the LLVM value being
1560  // privatized.
1561  for (auto [mlirPrivVar, mlirPrivatizerAttr] :
1562  llvm::zip_equal(mlirPrivVars, *mlirPrivSyms)) {
1563  // Find the MLIR private variable corresponding to the LLVM value
1564  // being privatized.
1565  llvm::Value *mlirToLLVMPrivVar =
1566  moduleTranslation.lookupValue(mlirPrivVar);
1567 
1568  // Check if the LLVM value being privatized matches the LLVM value
1569  // mapped to privVar. In some cases, this is not trivial ...
1570  auto isMatch = [&]() {
1571  if (mlirToLLVMPrivVar == nullptr)
1572  return false;
1573 
1574  // If both values are trivially equal, we found a match.
1575  if (mlirToLLVMPrivVar == &llvmOmpRegionInput)
1576  return true;
1577 
1578  // Otherwise, we check if both llvmOmpRegionInputPtr and
1579  // mlirToLLVMPrivVar refer to the same memory (through a load/store
1580  // pair). This happens if a struct (i.e. multi-field value) is being
1581  // privatized.
1582  //
1583  // For example, if the privatized value is defined by:
1584  // ```
1585  // %priv_val = alloca { ptr, i64 }, align 8
1586  // ```
1587  //
1588  // The initialization of this value (outside the omp region) will be
1589  // something like this:
1590  //
1591  // clang-format off
1592  // ```
1593  // %partially_init_priv_val = insertvalue { ptr, i64 } undef,
1594  // ptr %some_ptr, 0
1595  // %fully_init_priv_val = insertvalue { ptr, i64 } %partially_init_priv_val,
1596  // i64 %some_i64, 1
1597  // ...
1598  // store { ptr, i64 } %fully_init_priv_val, ptr %priv_val, align 8
1599  // ```
1600  // clang-format on
1601  //
1602  // Now, we enter the OMP region, in order to access this privatized
1603  // value, we need to load from the allocated memory:
1604  // ```
1605  // omp.par.entry:
1606  // %priv_val_load = load { ptr, i64 }, ptr %priv_val, align 8
1607  // ```
1608  //
1609  // The 2 LLVM values tracked here map as follows:
1610  // - `mlirToLLVMPrivVar` -> `%fully_init_priv_val`
1611  // - `llvmOmpRegionInputPtr` -> `%priv_val_load`
1612  //
1613  // Even though they eventually refer to the same memory reference
1614  // (the memory being privatized), they are 2 distinct LLVM values.
1615  // Therefore, we need to discover their correspondence by finding
1616  // out if they store into and load from the same mem ref.
1617  auto *llvmOmpRegionInputPtrLoad =
1618  llvm::dyn_cast_if_present<llvm::LoadInst>(&llvmOmpRegionInput);
1619 
1620  if (llvmOmpRegionInputPtrLoad == nullptr)
1621  return false;
1622 
1623  for (auto &use : mlirToLLVMPrivVar->uses()) {
1624  auto *mlirToLLVMPrivVarStore =
1625  llvm::dyn_cast_if_present<llvm::StoreInst>(use.getUser());
1626  if (mlirToLLVMPrivVarStore &&
1627  (llvmOmpRegionInputPtrLoad->getPointerOperand() ==
1628  mlirToLLVMPrivVarStore->getPointerOperand()))
1629  return true;
1630  }
1631 
1632  return false;
1633  };
1634 
1635  if (!isMatch())
1636  continue;
1637 
1638  SymbolRefAttr privSym = llvm::cast<SymbolRefAttr>(mlirPrivatizerAttr);
1639  omp::PrivateClauseOp privatizer = findPrivatizer(opInst, privSym);
1640 
1641  // Clone the privatizer in case it is used by more than one parallel
1642  // region. The privatizer is processed in-place (see below) before it
1643  // gets inlined in the parallel region and therefore processing the
1644  // original op is dangerous.
1645  return {mlirPrivVar,
1646  clonePrivatizer(moduleTranslation, privatizer, opInst)};
1647  }
1648  }
1649 
1650  return {mlir::Value(), omp::PrivateClauseOp()};
1651  }();
1652 
1653  if (mlirPrivVar) {
1654  Region &allocRegion = mlirPrivatizerClone.getAllocRegion();
1655 
1656  // If this is a `firstprivate` clause, prepare the `omp.private` op by:
1657  if (mlirPrivatizerClone.getDataSharingType() ==
1658  omp::DataSharingClauseType::FirstPrivate) {
1659  auto oldAllocBackBlock = std::prev(allocRegion.end());
1660  omp::YieldOp oldAllocYieldOp =
1661  llvm::cast<omp::YieldOp>(oldAllocBackBlock->getTerminator());
1662 
1663  Region &copyRegion = mlirPrivatizerClone.getCopyRegion();
1664 
1665  mlir::IRRewriter copyCloneBuilder(&moduleTranslation.getContext());
1666  // 1. Cloning the `copy` region to the end of the `alloc` region.
1667  copyCloneBuilder.cloneRegionBefore(copyRegion, allocRegion,
1668  allocRegion.end());
1669 
1670  auto newCopyRegionFrontBlock = std::next(oldAllocBackBlock);
1671  // 2. Merging the last `alloc` block with the first block in the `copy`
1672  // region clone.
1673  // 3. Re-mapping the first argument of the `copy` region to be the
1674  // argument of the `alloc` region and the second argument of the `copy`
1675  // region to be the yielded value of the `alloc` region (this is the
1676  // private clone of the privatized value).
1677  copyCloneBuilder.mergeBlocks(
1678  &*newCopyRegionFrontBlock, &*oldAllocBackBlock,
1679  {allocRegion.getArgument(0), oldAllocYieldOp.getOperand(0)});
1680 
1681  // 4. The old terminator of the `alloc` region is not needed anymore, so
1682  // delete it.
1683  oldAllocYieldOp.erase();
1684  }
1685

1687  // This way, the body of the privatizer will be changed from using the
1688  // region/block argument to the value being privatized.
1689  auto allocRegionArg = allocRegion.getArgument(0);
1690  replaceAllUsesInRegionWith(allocRegionArg, mlirPrivVar, allocRegion);

1693  builder.restoreIP(allocaIP);

1696  if (failed(inlineConvertOmpRegions(allocRegion, "omp.privatizer", builder,
1697  moduleTranslation, &yieldedValues))) {
1698  opInst.emitError("failed to inline `alloc` region of an `omp.private` "
1699  "op in the parallel region");

1709  mlirPrivatizerClones.push_back(mlirPrivatizerClone);

1712  builder.restoreIP(oldIP);
1713  }
1714

1716  };

1728  [](omp::DeclareReductionOp reductionDecl) {
1729  return &reductionDecl.getCleanupRegion();

1732  reductionCleanupRegions, privateReductionVariables,
1733  moduleTranslation, builder, "omp.reduction.cleanup")))
1734  bodyGenStatus = failure();

1737  llvm::transform(mlirPrivatizerClones,
1738  std::back_inserter(privateCleanupRegions),
1739  [](omp::PrivateClauseOp privatizer) {
1740  return &privatizer.getDeallocRegion();

1744  privateCleanupRegions, llvmPrivateVars, moduleTranslation, builder,
1745  "omp.private.dealloc", /*shouldLoadCleanupRegionArg=*/false)))
1746  bodyGenStatus = failure();

1748  builder.restoreIP(oldIP);
1749  };

1752  if (auto ifVar = opInst.getIfExpr())
1753  ifCond = moduleTranslation.lookupValue(ifVar);
1754  llvm::Value *numThreads = nullptr;
1755  if (auto numThreadsVar = opInst.getNumThreads())

1790 static LogicalResult
1791 convertOmpSimd(Operation &opInst, llvm::IRBuilderBase &builder,
1792  LLVM::ModuleTranslation &moduleTranslation) {
1793  auto simdOp = cast<omp::SimdOp>(opInst);
1794  auto loopOp = cast<omp::LoopNestOp>(simdOp.getWrappedLoop());
1795 
1796  if (!simdOp.getLinearVars().empty() || !simdOp.getLinearStepVars().empty() ||
1797  !simdOp.getPrivateVars().empty() || simdOp.getPrivateSyms() ||
1798  !simdOp.getReductionVars().empty() || simdOp.getReductionByref() ||
1799  simdOp.getReductionSyms())
1800  return opInst.emitError("unhandled clauses for translation to LLVM IR");
1801 
1802  llvm::OpenMPIRBuilder::LocationDescription ompLoc(builder);
1803 
1804  // Generator of the canonical loop body.
1805  // TODO: support error propagation in OpenMPIRBuilder and use it instead of
1806  // relying on captured variables.
1807  SmallVector<llvm::CanonicalLoopInfo *> loopInfos;
1808  SmallVector<llvm::OpenMPIRBuilder::InsertPointTy> bodyInsertPoints;
1809  LogicalResult bodyGenStatus = success();
1810  auto bodyGen = [&](llvm::OpenMPIRBuilder::InsertPointTy ip, llvm::Value *iv) {
1811  // Make sure further conversions know about the induction variable.
1812  moduleTranslation.mapValue(
1813  loopOp.getRegion().front().getArgument(loopInfos.size()), iv);
1814 
1815  // Capture the body insertion point for use in nested loops. BodyIP of the
1816  // CanonicalLoopInfo always points to the beginning of the entry block of
1817  // the body.
1818  bodyInsertPoints.push_back(ip);
1819 
1820  if (loopInfos.size() != loopOp.getNumLoops() - 1)
1821  return;
1822 
1823  // Convert the body of the loop.
1824  builder.restoreIP(ip);
1825  convertOmpOpRegions(loopOp.getRegion(), "omp.simd.region", builder,
1826  moduleTranslation, bodyGenStatus);
1827  };
1828 
1829  // Delegate actual loop construction to the OpenMP IRBuilder.
1830  // TODO: this currently assumes omp.loop_nest is semantically similar to SCF
1831  // loop, i.e. it has a positive step, uses signed integer semantics.
1832  // Reconsider this code when the nested loop operation clearly supports more
1833  // cases.
1834  llvm::OpenMPIRBuilder *ompBuilder = moduleTranslation.getOpenMPBuilder();
1835  for (unsigned i = 0, e = loopOp.getNumLoops(); i < e; ++i) {
1836  llvm::Value *lowerBound =
1837  moduleTranslation.lookupValue(loopOp.getLoopLowerBounds()[i]);
1838  llvm::Value *upperBound =
1839  moduleTranslation.lookupValue(loopOp.getLoopUpperBounds()[i]);
1840  llvm::Value *step = moduleTranslation.lookupValue(loopOp.getLoopSteps()[i]);
1841 
1842  // Make sure loop trip count are emitted in the preheader of the outermost
1843  // loop at the latest so that they are all available for the new collapsed
1844  // loop will be created below.
1845  llvm::OpenMPIRBuilder::LocationDescription loc = ompLoc;
1846  llvm::OpenMPIRBuilder::InsertPointTy computeIP = ompLoc.IP;
1847  if (i != 0) {
1848  loc = llvm::OpenMPIRBuilder::LocationDescription(bodyInsertPoints.back(),
1849  ompLoc.DL);
1850  computeIP = loopInfos.front()->getPreheaderIP();
1851  }
1852  loopInfos.push_back(ompBuilder->createCanonicalLoop(
1853  loc, bodyGen, lowerBound, upperBound, step,
1854  /*IsSigned=*/true, /*Inclusive=*/true, computeIP));
1855 
1856  if (failed(bodyGenStatus))
1857  return failure();
1858  }
1859 
1860  // Collapse loops.
1861  llvm::IRBuilderBase::InsertPoint afterIP = loopInfos.front()->getAfterIP();
1862  llvm::CanonicalLoopInfo *loopInfo =
1863  ompBuilder->collapseLoops(ompLoc.DL, loopInfos, {});
1864 
1865  llvm::ConstantInt *simdlen = nullptr;
1866  if (std::optional<uint64_t> simdlenVar = simdOp.getSimdlen())
1867  simdlen = builder.getInt64(simdlenVar.value());
1868 
1869  llvm::ConstantInt *safelen = nullptr;
1870  if (std::optional<uint64_t> safelenVar = simdOp.getSafelen())
1871  safelen = builder.getInt64(safelenVar.value());
1872 
1873  llvm::MapVector<llvm::Value *, llvm::Value *> alignedVars;
1874  llvm::omp::OrderKind order = convertOrderKind(simdOp.getOrder());
1875  ompBuilder->applySimd(loopInfo, alignedVars,
1876  simdOp.getIfExpr()
1877  ? moduleTranslation.lookupValue(simdOp.getIfExpr())
1878  : nullptr,
1879  order, simdlen, safelen);
1880 
1881  builder.restoreIP(afterIP);
1882  return success();
1883 }

2301 // This function calculates the size to be offloaded for a specified type, given
2302 // its associated map clause (which can contain bounds information which affects
2303 // the total size), this size is calculated based on the underlying element type
2304 // e.g. given a 1-D array of ints, we will calculate the size from the integer
2305 // type * number of elements in the array. This size can be used in other
2306 // calculations but is ultimately used as an argument to the OpenMP runtimes
2307 // kernel argument structure which is generated through the combinedInfo data
2308 // structures.
2309 // This function is somewhat equivalent to Clang's getExprTypeSize inside of
2310 // CGOpenMPRuntime.cpp.
2311 llvm::Value *getSizeInBytes(DataLayout &dl, const mlir::Type &type,
2312  Operation *clauseOp, llvm::Value *basePointer,
2313  llvm::Type *baseType, llvm::IRBuilderBase &builder,
2314  LLVM::ModuleTranslation &moduleTranslation) {
2315  if (auto memberClause =
2316  mlir::dyn_cast_if_present<mlir::omp::MapInfoOp>(clauseOp)) {
2317  // This calculates the size to transfer based on bounds and the underlying
2318  // element type, provided bounds have been specified (Fortran
2319  // pointers/allocatables/target and arrays that have sections specified fall
2320  // into this as well).
2321  if (!memberClause.getBounds().empty()) {
2322  llvm::Value *elementCount = builder.getInt64(1);
2323  for (auto bounds : memberClause.getBounds()) {
2324  if (auto boundOp = mlir::dyn_cast_if_present<mlir::omp::MapBoundsOp>(
2325  bounds.getDefiningOp())) {
2326  // The below calculation for the size to be mapped calculated from the

2328  // multiply by the underlying element types byte size to get the full
2329  // size to be offloaded based on the bounds
2330  elementCount = builder.CreateMul(
2331  elementCount,
2332  builder.CreateAdd(
2333  builder.CreateSub(
2334  moduleTranslation.lookupValue(boundOp.getUpperBound()),
2335  moduleTranslation.lookupValue(boundOp.getLowerBound())),
2336  builder.getInt64(1)));
2337  }
2338  }
2339 
2340  // utilising getTypeSizeInBits instead of getTypeSize as getTypeSize gives
2341  // the size in inconsistent byte or bit format.
2342  uint64_t underlyingTypeSzInBits = dl.getTypeSizeInBits(type);
2343  if (auto arrTy = llvm::dyn_cast_if_present<LLVM::LLVMArrayType>(type))
2344  underlyingTypeSzInBits = getArrayElementSizeInBits(arrTy, dl);
2345 
2346  // The size in bytes x number of elements, the sizeInBytes stored is
2347  // the underyling types size, e.g. if ptr<i32>, it'll be the i32's

2349  // bytes from the extent (ub - lb) * sizeInBytes. NOTE: This may need

2351  return builder.CreateMul(elementCount,
2352  builder.getInt64(underlyingTypeSzInBits / 8));

2519 /// This function calculates the array/pointer offset for map data provided
2520 /// with bounds operations, e.g. when provided something like the following:
2521 ///
2522 /// Fortran
2523 /// map(tofrom: array(2:5, 3:2))
2524 /// or
2525 /// C++
2526 /// map(tofrom: array[1:4][2:3])
2527 /// We must calculate the initial pointer offset to pass across, this function
2528 /// performs this using bounds.
2529 ///
2530 /// NOTE: which while specified in row-major order it currently needs to be
2531 /// flipped for Fortran's column order array allocation and access (as
2532 /// opposed to C++'s row-major, hence the backwards processing where order is
2533 /// important). This is likely important to keep in mind for the future when
2534 /// we incorporate a C++ frontend, both frontends will need to agree on the
2535 /// ordering of generated bounds operations (one may have to flip them) to
2536 /// make the below lowering frontend agnostic. The offload size
2537 /// calcualtion may also have to be adjusted for C++.
2538 std::vector<llvm::Value *>
2539 calculateBoundsOffset(LLVM::ModuleTranslation &moduleTranslation,
2540  llvm::IRBuilderBase &builder, bool isArrayTy,
2541  OperandRange bounds) {
2542  std::vector<llvm::Value *> idx;
2543  // There's no bounds to calculate an offset from, we can safely
2544  // ignore and return no indices.
2545  if (bounds.empty())
2546  return idx;
2547 
2548  // If we have an array type, then we have its type so can treat it as a
2549  // normal GEP instruction where the bounds operations are simply indexes
2550  // into the array. We currently do reverse order of the bounds, which
2551  // I believe leans more towards Fortran's column-major in memory.
2552  if (isArrayTy) {
2553  idx.push_back(builder.getInt64(0));
2554  for (int i = bounds.size() - 1; i >= 0; --i) {
2555  if (auto boundOp = dyn_cast_if_present<omp::MapBoundsOp>(
2556  bounds[i].getDefiningOp())) {
2557  idx.push_back(moduleTranslation.lookupValue(boundOp.getLowerBound()));
2558  }
2559  }
2560  } else {
2561  // If we do not have an array type, but we have bounds, then we're dealing
2562  // with a pointer that's being treated like an array and we have the
2563  // underlying type e.g. an i32, or f64 etc, e.g. a fortran descriptor base
2564  // address (pointer pointing to the actual data) so we must caclulate the
2565  // offset using a single index which the following two loops attempts to
2566  // compute.
2567 
2568  // Calculates the size offset we need to make per row e.g. first row or
2569  // column only needs to be offset by one, but the next would have to be
2570  // the previous row/column offset multiplied by the extent of current row.
2571  //
2572  // For example ([1][10][100]):
2573  //
2574  // - First row/column we move by 1 for each index increment
2575  // - Second row/column we move by 1 (first row/column) * 10 (extent/size of
2576  // current) for 10 for each index increment
2577  // - Third row/column we would move by 10 (second row/column) *
2578  // (extent/size of current) 100 for 1000 for each index increment
2579  std::vector<llvm::Value *> dimensionIndexSizeOffset{builder.getInt64(1)};
2580  for (size_t i = 1; i < bounds.size(); ++i) {
2581  if (auto boundOp = dyn_cast_if_present<omp::MapBoundsOp>(
2582  bounds[i].getDefiningOp())) {
2583  dimensionIndexSizeOffset.push_back(builder.CreateMul(
2584  moduleTranslation.lookupValue(boundOp.getExtent()),
2585  dimensionIndexSizeOffset[i - 1]));
2586  }
2587  }
2588 
2589  // Now that we have calculated how much we move by per index, we must
2590  // multiply each lower bound offset in indexes by the size offset we
2591  // have calculated in the previous and accumulate the results to get
2592  // our final resulting offset.
2593  for (int i = bounds.size() - 1; i >= 0; --i) {
2594  if (auto boundOp = dyn_cast_if_present<omp::MapBoundsOp>(
2595  bounds[i].getDefiningOp())) {
2596  if (idx.empty())
2597  idx.emplace_back(builder.CreateMul(
2598  moduleTranslation.lookupValue(boundOp.getLowerBound()),
2599  dimensionIndexSizeOffset[i]));
2600  else
2601  idx.back() = builder.CreateAdd(
2602  idx.back(), builder.CreateMul(moduleTranslation.lookupValue(
2603  boundOp.getLowerBound()),
2604  dimensionIndexSizeOffset[i]));
2605  }
2606  }
2607  }
2608 
2609  return idx;
2610 }

2849 // This is a variation on Clang's GenerateOpenMPCapturedVars, which
2850 // generates different operation (e.g. load/store) combinations for
2851 // arguments to the kernel, based on map capture kinds which are then
2852 // utilised in the combinedInfo in place of the original Map value.
2853 static void
2854 createAlteredByCaptureMap(MapInfoData &mapData,
2855  LLVM::ModuleTranslation &moduleTranslation,
2856  llvm::IRBuilderBase &builder) {
2857  for (size_t i = 0; i < mapData.MapClause.size(); ++i) {
2858  // if it's declare target, skip it, it's handled separately.
2859  if (!mapData.IsDeclareTarget[i]) {
2860  auto mapOp = cast<omp::MapInfoOp>(mapData.MapClause[i]);

2862  mapOp.getMapCaptureType().value_or(omp::VariableCaptureKind::ByRef);
2863  bool isPtrTy = checkIfPointerMap(mapOp);
2864

2866  // logic may be required in the future. Clang invokes EmitLValue,
2867  // which has specialised logic for special Clang types such as user
2868  // defines, so it is possible we will have to extend this for

2870  // function mimics some of the logic from Clang that we require for

2902  newV = builder.CreateLoad(builder.getPtrTy(), memTempAlloc);

2906  mapData.BasePointers[i] = newV;

3315 // instructions for retrieving (acessing) kernel
3316 // arguments inside of the device kernel for use by
3317 // the kernel. This enables different semantics such as
3318 // the creation of temporary copies of data allowing
3319 // semantics like read-only/no host write back kernel
3320 // arguments.
3321 //
3322 // This currently implements a very light version of Clang's
3323 // EmitParmDecl's handling of direct argument handling as well
3324 // as a portion of the argument access generation based on
3325 // capture types found at the end of emitOutlinedFunctionPrologue
3326 // in Clang. The indirect path handling of EmitParmDecl's may be
3327 // required for future work, but a direct 1-to-1 copy doesn't seem
3328 // possible as the logic is rather scattered throughout Clang's
3329 // lowering and perhaps we wish to deviate slightly.
